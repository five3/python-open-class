# Python模块化 - 第二节 类 

### 定义
```python
class Foo(object):
    def __init__(self, name):
        self.name = name
        
    def say(self, what='hello'):
        print('%s %s' % (what, self.name))
        
    def __cry(self):
        print('crying')
                
    def __del__(self):
        print('i am in del')
```

### 类的实例化
```python
foo = Foo('python')
foo.say()   # hello python
foo.say('fuck')    # fuck python  
foo._Foo__cry()   # cry
del foo
```

### 静态方法
```python
class Foo(object):
    @staticmethod
    def say(what):
        print(f'{what}')
        
Foo.say('python')
```

### 类方法和属性
```python
class Foo(object):
    alisa = 'foo'
    
    @classmethod
    def say(cls, what):
        print('say {1} in {0}, alisa is {2}'.format(cls.__name__, what, cls.alisa))
        
Foo.say('hello')
print(Foo.alisa)
```

### 实例属性
```python
class Foo(object):
    def __init__(self, name):
        self._name = name

    @property
    def name(self):
       return '{what}'.format(what=self._name)

    @name.setter
    def name(self, value):
        if isinstance(value, bytes):
            self._name = value
        else:
            self._name = None
            
f = Foo('hello')
print(f.name)
f.name = 22
print(f.name)
f._name = 2
print(f.name)
```

### 类继承
#### 只继承不覆盖
```python
class base(object):
    def __init__(self):
        print('in base init')

    def say(self):
        print('in base say')

    def __del__(self):
        print('in base del')

class father(base):
    pass

f = father()
f.say()
del f
```

#### 继承并覆盖
```python
class base(object):
    def __init__(self):
        print('in base init')

    def say(self):
        print('in base say')

    def __del__(self):
        print('in base del')

class father(base):
    def __init__(self):
        print('in father init')

    def say(self):
        print('in father say')

    def __del__(self):
        print('in father del')

f = father()
f.say()
del f
```

#### 覆盖并调用父类
```python
class base(object):
    def __init__(self):
        print('in base init')

    def say(self):
        print('in base say')

    def __del__(self):
        print('in base del')

class father(base):
    def __init__(self):
        super().__init__()
        print('in father init')

    def say(self):
        super().say()
        print('in father say')

    def __del__(self):
        super().__del__()
        print('in father del')

f = father()
f.say()
del f
```

#### 多重继承
```python
class Foo(object):
    def say(self):
        print('in say')

class Bar(object):
    def eat(self):
        print('in eat')

class Baz(Foo, Bar):
    pass

b = Baz()
b.say()
b.eat()
```

#### 多重继承的顺序
```python
class Foo(object):
    def say(self):
        print('in say')

    def run(self):
        print('run in', self.__class__.__name__)

class Bar(object):
    def eat(self):
        print('in eat')

    def run(self):
        print('run in', self.__class__.__name__)

class Baz(Bar, Foo):
    pass

b = Baz()
b.run()
print(Baz.__mro__)
```
